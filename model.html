import keras.backend as K
'''
MSE, only over masked areas
'''
def masked_MSE(y_true, y_pred):
  for yt in y_true: # for each example in the batch
    yt = yt[mask == True]
  for yp in y_pred:
    yp = yp[mask == True]
  loss_val = K.mean(K.square(y_pred - y_true))
  return loss_val
